# 单机版安装
## 前期准备
```bash
sudo hostnamectl set-hostname K8S1

cat >> /etc/hosts <<EOF
192.168.1.158 K8S1
EOF

# 关闭防火墙和selinux
sudo systemctl stop firewalld
sudo systemctl disable firewalld
sudo sed -i 's/SELINUX=enforcing/SELINUX=disabled/' /etc/selinux/config
sudo setenforce 0

# 关闭swap
```




```bash
# 初始化
kubeadm init --image-repository=registry.cn-hangzhou.aliyuncs.com/google_containers --apiserver-advertise-address=192.168.1.158 --kubernetes-version=v1.25.4  --service-cidr=10.15.0.0/16--pod-network-cidr=10.18.0.0/16
```
